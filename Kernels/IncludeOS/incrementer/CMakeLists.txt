cmake_minimum_required(VERSION 3.0)
set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)
project (incrementer)
include(${CMAKE_CURRENT_BINARY_DIR}/conanbuildinfo.cmake OPTIONAL RESULT_VARIABLE HAS_CONAN)
conan_basic_setup()
include(os)

os_add_executable(incrementer "Get a Timestamp for Now" main.cpp)
os_add_drivers(incrementer virtionet vmxnet3 e1000 boot_logger)
os_add_plugins(incrementer autoconf)
os_add_stdout(incrementer default_stdout)
os_add_ssl_certificates(incrementer)
